CREATE PROCEDURE `CheckPembayaran`(IN `param_kodepemesanan` VARCHAR(255), IN `param_idkegiatan` VARCHAR(255)) DETERMINISTIC NO SQL SQL SECURITY DEFINER SELECT 
    CASE 
        WHEN SUM(a.Total) <= IFNULL((SELECT SUM(Nominal) FROM `bk_bukti_pembayaran` WHERE KodePemesanan = param_kodepemesanan AND IdKegiatan = param_idkegiatan), 0)
            THEN 1
        ELSE 0
    END AS Status,
    (
		SELECT StatusKonfirmasi FROM booking_acara WHERE TipePendaftar = 'Pendaftar' AND KodePemesanan = param_kodepemesanan AND IdKegiatan = param_idkegiatan
    ) AS StatusKonfirmasi,
 	a.Total,
    IFNULL((SELECT SUM(Nominal) FROM `bk_bukti_pembayaran` WHERE KodePemesanan = param_kodepemesanan AND IdKegiatan = param_idkegiatan), 0) AS Bayar,
 	SUM(a.Total) - IFNULL((SELECT SUM(Nominal) FROM `bk_bukti_pembayaran` WHERE KodePemesanan = param_kodepemesanan AND IdKegiatan = param_idkegiatan), 0) Sisa
FROM 
(
    SELECT 
        bo.KodePemesanan, IFNULL(bp.Harga, bo.NominalDonasi) Harga, bo.Quantity, IFNULL(bp.Harga, bo.NominalDonasi) * bo.Quantity AS Total
        FROM bk_order bo 
    LEFT JOIN bk_product bp ON bp.Id = bo.IdProduct 
    INNER JOIN booking_acara ba ON ba.Id = bo.IdBooking
    WHERE bo.RecordStatus != 'D' AND bo.KodePemesanan = param_kodepemesanan AND ba.IdKegiatan = param_idkegiatan
) a

CREATE PROCEDURE `getKehadiranAcara`(IN `param_idsubweb` VARCHAR(255), IN `param_idkegiatan` VARCHAR(255), OUT `TotalL` INT(255), OUT `DipesanL` INT(255), OUT `HadirL` INT(255), OUT `TotalP` INT(255), OUT `HadirP` INT(255), OUT `DipesanP` INT(255)) NOT DETERMINISTIC NO SQL SQL SECURITY DEFINER BEGIN

SET DipesanP = 0;

SELECT MaxP INTO TotalP FROM bk_ruangan WHERE IdKegiatan = param_idkegiatan;

-- laki - laki
SELECT COUNT(ba.Id) INTO DipesanL
    FROM booking_acara ba
WHERE 
    ba.JenisKelamin IS NOT NULL AND 
    ba.Kehadiran = 'HP' AND 
    ba.StatusKonfirmasi != 'D' AND

    ba.IdKegiatan = param_idkegiatan AND 
    ba.JenisKelamin = 'L' AND
    ba.CountedKapasitas = 1;

SELECT COUNT(ba.Id) INTO DipesanP
    FROM booking_acara ba

    ba.JenisKelamin IS NOT NULL AND 
    ba.Kehadiran = 'HP' AND 
    ba.StatusKonfirmasi != 'D' AND 
    ba.RecordStatus != 'D' AND
    ba.IdSubweb = param_idsubweb AND 
    ba.IdKegiatan = param_idkegiatan AND 
    ba.JenisKelamin = 'P' AND
    ba.CountedKapasitas = 1;

-- total hadir
-- laki - laki
SELECT COUNT(ua.IdKegiatan) INTO HadirL FROM (
    SELECT DISTINCT IdUser, IdKegiatan
        FROM booking_acara ba
        WHERE 
            ba.JenisKelamin IS NOT NULL AND 
            ba.Kehadiran = 'HP' AND 
            ba.StatusKonfirmasi != 'D' AND 
            ba.RecordStatus != 'D' AND
            ba.IdSubweb = param_idsubweb AND 
            ba.IdKegiatan = param_idkegiatan AND 
            ba.JenisKelamin = 'L' AND
            ba.CountedKapasitas = 1
    ) baa
INNER JOIN user_activity ua ON baa.IdUser = ua.IdUser AND baa.IdKegiatan = param_idkegiatan AND ua.IdKegiatan = param_idkegiatan;

-- perempuan
SELECT COUNT(ua.IdKegiatan) INTO HadirP FROM (
    SELECT DISTINCT IdUser, IdKegiatan
        FROM booking_acara ba

    ) baa
INNER JOIN user_activity ua ON baa.IdUser = ua.IdUser AND baa.IdKegiatan = ua.IdKegiatan AND ua.IdKegiatan = param_idkegiatan;

SELECT Judul INTO NamaKegiatan FROM kalender_kegiatan WHERE Id = param_idkegiatan;
